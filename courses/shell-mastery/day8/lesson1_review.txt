## The Professional Script Template
  #!/bin/bash
  set -euo pipefail

  SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
  source "$SCRIPT_DIR/lib/utils.sh"

  trap cleanup EXIT
  trap 'die "Interrupted"' INT TERM
  trap 'die "Error at line $LINENO"' ERR

  main() {
      parse_args "$@"
      validate_inputs
      acquire_lock
      log INFO "Starting..."
      do_work
      log INFO "Done."
  }

  main "$@"

Every production script follows this shape.
